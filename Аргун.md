(
    supplier_id uuid references supplier_erq,
    product_id uuid references product_yqn,

);

(
    id                uuid primary key,
    model             text,
    body_volume       int,
    lifting_capacity  int
);

create table supplier_vehicle_info_vgq
(
    supplier_id uuid references supplier_erq,

);

-- 6. Вывести модели электронных книг с наименьшим весом и наименьшей ценой среди электронных книг с наименьшим весом.
WITH min_weight_books AS (
    SELECT
        id AS product_id,
        model,
        weight
    FROM
        product_yqn
    WHERE
        weight = (SELECT MIN(weight) FROM product_yqn)
),
min_weight_min_price_books AS (
    SELECT
        p.model,
        p.weight,
        spi.price
    FROM
        min_weight_books mwb
    INNER JOIN
        supplier_product_info_o2i spi ON mwb.product_id = spi.product_id
    INNER JOIN
        product_yqn p ON mwb.product_id = p.id
    WHERE
        spi.price = (SELECT MIN(price) FROM supplier_product_info_o2i WHERE product_id IN (SELECT id FROM min_weight_books))
)
SELECT
    model,
    weight,
    price
FROM
    min_weight_min_price_books;
